# Git
Connect VSCode - Git - GitHub

# 1. git - add, commit
git init  
git add -A  
git commit -m ""  

# 2. bracn 생성, github push
git branch -M master(브랜치이름)  
git branch (브랜치 생성 확인)  
git remote add origin(원격이름) (github주소)  
git push -u origin(원격이름) master(브랜치이름)

# 3. 깃허브에서 가져오기
git clone "github주소"  
복사, 폴더생김  
cd 생긴 폴더위치로 이동  
작업 후  
git add  
git commit
